**Embedded vs Referencing (VERY IMPORTANT INTERVIEW POINT)**
| Feature           | Embedded     | Referencing     |
| ----------------- | ------------ | --------------- |

| Write flexibility | ❌ Limited    | ✅ Flexible      |
| Data duplication  | ❌ Yes        | ✅ No            |
| Best use case     | Orders, logs | Users, products |

**When to use Embedded**
* Order + order_items (ALWAYS embedded in real systems)

**When to use Referencing**
* User, Product (shared data)

db.orders.find({
  created_at: { $gt: new Date("2026-03-27") }

//22. Find products not in stock (stock = 0).

  stock: 0

//23. Get orders with total between 3000 and 20000.
db.orders.find({
  total: { $gte: 3000, $lte: 20000 }

/*
**Key MongoDB Query Operators (VERY IMPORTANT)**
| Operator | Meaning             |
| -------- | ------------------- |
| `$gt`    | Greater than        |
| `$lt`    | Less than           |
| `$gte`   | ≥                   |
| `$lte`   | ≤                   |
| `$regex` | Pattern matching    |
| `$and`   | Multiple conditions |
| `$or`    | Either condition    |

Example (Advanced Filter)
db.products.find({
  $and: [
    { price: { $gt: 1000 } },
    { stock: { $lt: 80 } }
  ]
})

Real-World Tips
* Always use **projection** → reduces data transfer
* Use **indexes (later phase)** for these queries
* Regex queries can be slow without indexing
*/

//-- -

//24. Calculate average product price.
db.products.aggregate([
  {
    $group: {
      _id: null,
      avgPrice: { $avg: "$price" }
    }
  }
])

//25. Find maximum and minimum product price.
db.products.aggregate([
  {
    $group: {
      _id: null,
      maxPrice: { $max: "$price" },
      minPrice: { $min: "$price" }
    }
  }
])

//26. Convert all user names to uppercase using aggregation.
db.users.aggregate([
  {
    $project: {
      _id: 0,
      name: { $toUpper: "$name" },
      email: 1
    }
  }
])

//27. Extract date (YYYY-MM-DD) from order created_at.
db.orders.aggregate([
  {
    $project: {
      order_date: {
        $dateToString: {
          format: "%Y-%m-%d",
          date: "$created_at"
        }
      }
    }
  }
])

//28. Count total number of products.
db.products.aggregate([
  {
    $count: "total_products"
  }
])

//29. Find total orders per user.
db.orders.aggregate([
  {
    $group: {
      _id: "$user_id",
      total_orders: { $sum: 1 }
    }
  }
])

//30. Calculate total revenue generated per user.
db.orders.aggregate([
  {
    $group: {
      _id: "$user_id",
      total_revenue: { $sum: "$total" }
    }
  }
])

//31. Find number of products sold per product.
// Important: Uses **unwind (very important concept)**
db.orders.aggregate([
  { $unwind: "$order_items" },
  {
    $group: {
      _id: "$order_items.product_id",
      total_quantity: { $sum: "$order_items.quantity" }
    }
  }
])

//32. Get users who placed more than 1 order.
db.orders.aggregate([
  {
    $group: {
      _id: "$user_id",
      total_orders: { $sum: 1 }
    }
  },
  {
    $match: {
      total_orders: { $gt: 1 }
    }
  }
])

//33. Find products with total sales quantity > 1.
db.orders.aggregate([
  { $unwind: "$order_items" },
  {
    $group: {
      _id: "$order_items.product_id",
      total_sold: { $sum: "$order_items.quantity" }
    }
  },
  {
    $match: {
      total_sold: { $gt: 1 }
    }
  }
])
/*
**Core Aggregation Concepts (CRITICAL)**
    `$group`Like SQL `GROUP BY`
    `$project` Select / transform fields
    `$unwind` Break array into multiple documents
        Example:
            order_items: [
            { product_id: 1 },
            { product_id: 3 }
            ]
            ➡ becomes:
            { product_id: 1 }
            { product_id: 3 }
    `$match` Filter (like WHERE)
    `$count` Count documents
Real Interview Tip
    Most important pipeline pattern: $unwind → $group → $match → $project
*/

//-- -

//34. Retrieve orders along with user details.
// Join: `orders → users`
db.orders.aggregate([
  {
    $lookup: {
      from: "users",
      localField: "user_id",
      foreignField: "_id",
      as: "user_details"
    }
  },
  { $unwind: "$user_details" }
])

//35. Display order details with product names.
// Join: `orders → products`
db.orders.aggregate([
  { $unwind: "$order_items" },
  {
    $lookup: {
      from: "products",
      localField: "order_items.product_id",
      foreignField: "_id",
      as: "product_details"
    }
  },
  { $unwind: "$product_details" },
  {
    $project: {
      _id: 1,
      product_name: "$product_details.name",
      quantity: "$order_items.quantity"
    }
  }
])

//36. Get all products and their order quantities (include products not ordered).
// LEFT JOIN behavior
db.products.aggregate([
  {
    $lookup: {
      from: "orders",
      let: { productId: "$_id" },
      pipeline: [
        { $unwind: "$order_items" },
        {
          $match: {
            $expr: {
              $eq: ["$order_items.product_id", "$$productId"]
            }
          }
        }
      ],
      as: "orders_info"
    }
  },
  {
    $project: {
      name: 1,
      total_ordered: {
        $sum: "$orders_info.order_items.quantity"
      }
    }
  }
])
// Even products with **no orders** will appear (important)

//37. Find users who have never placed an order.
db.users.aggregate([
  {
    $lookup: {
      from: "orders",
      localField: "_id",
      foreignField: "user_id",
      as: "orders"
    }
  },
  {
    $match: {
      orders: { $size: 0 }
    }
  }
])

//38. Retrieve full order summary: (user + product + quantity).
//  Real-world reporting query
db.orders.aggregate([
  { $unwind: "$order_items" },

  {
    $lookup: {
      from: "users",
      localField: "user_id",
      foreignField: "_id",
      as: "user"
    }
  },
  { $unwind: "$user" },

  {
    $lookup: {
      from: "products",
      localField: "order_items.product_id",
      foreignField: "_id",
      as: "product"
    }
  },
  { $unwind: "$product" },

  {
    $project: {
      order_id: "$_id",
      user_name: "$user.name",
      product_name: "$product.name",
      quantity: "$order_items.quantity"
    }
  }
])

/*
**Key Concept: $lookup (MongoDB JOIN)**
Syntax
{
  $lookup: {
    from: "collection",
    localField: "field",
    foreignField: "field",
    as: "output_array"
  }
}

Types of JOIN in MongoDB
| SQL        | MongoDB             |
| ---------- | ------------------- |
| INNER JOIN | `$lookup + $unwind` |
| LEFT JOIN  | `$lookup`           |
| RIGHT JOIN | Not directly        |
| FULL JOIN  | Manual              |

Performance Warning
* `$lookup` is **expensive**
* Prefer **embedding** when possible (orders → order_items)

Real-World Insight
    Best practice:
    * **Users, Products → referenced**
    * **Orders + items → embedded**
    * Use `$lookup` only for **reports / dashboards**
*/

//-- -

//39. Find users who placed orders above average order value.
// Step: 1. Find average order value 2. Filter users above that
db.orders.aggregate([
  {
    $group: {
      _id: null,
      avgOrderValue: { $avg: "$total" }
    }
  },
  {
    $lookup: {
      from: "orders",
      pipeline: [
        {
          $match: {
            $expr: {
              $gt: ["$total", "$$avg"]
            }
          }
        }
      ],
      as: "high_orders"
    }
  }
])

// Cleaner (practical approach):
let avg = db.orders.aggregate([
  { $group: { _id: null, avg: { $avg: "$total" } } }
]).toArray()[0].avg

db.orders.find({ total: { $gt: avg } })

//40. Get products whose price is greater than average price.
let avgPrice = db.products.aggregate([
  { $group: { _id: null, avg: { $avg: "$price" } } }
]).toArray()[0].avg

db.products.find({
  price: { $gt: avgPrice }
})

//41. Find the most expensive product purchased in any order.
db.orders.aggregate([
  { $unwind: "$order_items" },
  {
    $lookup: {
      from: "products",
      localField: "order_items.product_id",
      foreignField: "_id",
      as: "product"
    }
  },
  { $unwind: "$product" },
  {
    $sort: { "product.price": -1 }
  },
  { $limit: 1 }
])

//42. Find latest order per user.
db.orders.aggregate([
  { $sort: { user_id: 1, created_at: -1 } },
  {
    $group: {
      _id: "$user_id",
      latest_order: { $first: "$$ROOT" }
    }
  }
])

//43. Get top 3 most sold products.
db.orders.aggregate([
  { $unwind: "$order_items" },
  {
    $group: {
      _id: "$order_items.product_id",
      total_sold: { $sum: "$order_items.quantity" }
    }
  },
  { $sort: { total_sold: -1 } },
  { $limit: 3 }
])

//44. Find users who purchased more than 2 different products.
db.orders.aggregate([
  { $unwind: "$order_items" },
  {
    $group: {
      _id: "$user_id",
      unique_products: { $addToSet: "$order_items.product_id" }
    }
  },
  {
    $project: {
      count: { $size: "$unique_products" }
    }
  },
  {
    $match: {
      count: { $gt: 2 }
    }
  }
])

/*
**Advanced Concepts You Just Used**
    `$sort` + `$limit` Top-N queries (very common)
    `$addToSet` Unique values (like DISTINCT)
    `$first` Get latest/first record after sorting

Multi-step Aggregation Logic
    Real-world pattern: $unwind → $lookup → $group → $sort → $limit

Real Business Use Cases
| Query             | Use Case               |
| ----------------- | ---------------------- |
| Above avg orders  | Premium customers      |
| Top products      | Best sellers           |
| Latest order      | User activity tracking |
| Expensive product | Revenue insights       |
*/

/*
**Indexing Concepts (VERY IMPORTANT)**
Types of Indexes
| Type         | Use                              |
| ------------ | -------------------------------- |
| Single Field | `{ name: 1 }`                    |
| Compound     | `{ user_id: 1, created_at: -1 }` |
| Text         | Full-text search                 |
| Multikey     | Arrays                           |
| TTL          | Auto-delete                      |

Compound Index Rule
    Order matters!
    { user_id: 1, created_at: -1 }

✔ Works for:
* `user_id`
* `user_id + created_at`

❌ Not for:
* only `created_at`

Index Trade-offs
| Advantage         | Disadvantage  |
| ----------------- | ------------- |
| Faster reads      | Slower writes |
| Efficient queries | More storage  |

Real-World Best Practices
* Index fields used in:
  * `find()`
  * `sort()`
  * `$lookup`
* Avoid over-indexing ❌
* Always test with `.explain()`
*/

//49. Write a transaction to: Insert order + Reduce product stock
/*
**Important Prerequisite**
👉 Transactions work only in:
* Replica Set (even single-node replica set is fine)
* MongoDB 4.0+

Goal
* Insert order
* Reduce product stock
* Both must succeed together
*/
const session = db.getMongo().startSession()

session.startTransaction()

try {
  const orders = session.getDatabase("ecommerce_db").orders
  const products = session.getDatabase("ecommerce_db").products

  // Step 1: Insert Order
  orders.insertOne({
    _id: 7,
    user_id: 1,
    total: 17000,
    created_at: new Date(),
    order_items: [
      { product_id: 1, quantity: 1 },
      { product_id: 3, quantity: 1 }
    ],
    status: "placed"
  })

  // Step 2: Reduce Stock
  products.updateOne(

const session = db.getMongo().startSession()

session.startTransaction()

try {
  const dbRef = session.getDatabase("ecommerce_db")
  const products = dbRef.products
  const orders = dbRef.orders

  const items = [
    { product_id: 1, quantity: 2 },
    { product_id: 2, quantity: 1 }
  ]

  // Step 1: Check stock
  for (let item of items) {
    let product = products.findOne({ _id: item.product_id })

    if (!product || product.stock < item.quantity) {
      throw new Error("Insufficient stock for product: " + item.product_id)
    }
  }

  // Step 2: Insert order
  orders.insertOne({
    _id: 8,
    user_id: 2,
    total: 80000,
    created_at: new Date(),
    order_items: items,
    status: "placed"
  })

  // Step 3: Reduce stock
  for (let item of items) {
    products.updateOne(
      { _id: item.product_id },
      { $inc: { stock: -item.quantity } }
    )
  }

  // Commit
  session.commitTransaction()
  print("Transaction Successful ✅")
} catch (error) {
  session.abortTransaction()
  print("Transaction Failed & Rolled Back ❌")
  print(error.message)
}
