# Question 8
```
Develop two routes: find_prime_100 and find_cube_100 which prints prime numbers less than 100 and
cubes less than 100 using Express JS routing mechanism
```

```bash id="2c52j3"

```


---

# Step 3: Create Express Application

# server.js

```javascript id="k4l7h7"
const express = require('express');

const app = express();

const PORT = 3000;


// Route 1
// Find Prime Numbers Less Than 100

app.get('/find_prime_100', (req, res) => {

    let primes = [];

    for (let i = 2; i < 100; i++) {

        let isPrime = true;

        for (let j = 2; j <= Math.sqrt(i); j++) {

            if (i % j === 0) {

                isPrime = false;
                break;
            }
        }

        if (isPrime) {
            primes.push(i);
        }
    }

    res.send(
        "<h2>Prime Numbers Less Than 100</h2>" +
        primes.join(", ")
    );
});


// Route 2
// Find Cubes Less Than 100

app.get('/find_cube_100', (req, res) => {

    let cubes = [];

    let i = 1;

    while (true) {

        let cube = i * i * i;

        if (cube >= 100)
            break;

        cubes.push(cube);

        i++;
    }

    res.send(
        "<h2>Cubes Less Than 100</h2>" +
        cubes.join(", ")
    );
});


// Start Server

app.listen(PORT, () => {

    console.log(
        `Server running at http://localhost:${PORT}`
    );
});
```
