# backend/server.js

```javascript id="jlwmdq"
const express = require('express');

const mongoose = require('mongoose');

const cors = require('cors');

const app = express();

const PORT = 3000;

app.use(cors());

app.use(express.json());



// MongoDB Connection

mongoose.connect(
    'mongodb://127.0.0.1:27017/airline'
)

.then(() => {

    console.log("MongoDB Connected");
})

.catch((err) => {

    console.log(err);
});




// Flight Schema

const flightSchema = new mongoose.Schema({

    flight_name: String,

    source: String,

    destination: String,

    price: Number
});




// Flight Model

const Flight = mongoose.model(
    'flights',
    flightSchema
);




// REST API

app.get('/flights', async (req, res) => {

    try {

        const flights = await Flight.find();

        res.json(flights);

    } catch (err) {

        res.status(500).json({
            message: err.message
        });
    }
});




app.listen(PORT, () => {

    console.log(
        `Server running at http://localhost:${PORT}`
    );
});
```
