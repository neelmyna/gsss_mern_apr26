```
Develop the application that sends fruit name and price data from client side to Node.js server using Ajax
```

# Step 3: Create Node.js Server

# server.js

```javascript id="nn2lgo"
const express = require('express');

const app = express();

const PORT = 3000;

// Middleware
app.use(express.json());

// Serve static files
app.use(express.static('public'));

// POST request handler
app.post('/fruit', (req, res) => {

    const fruitName = req.body.fruit;
    const price = req.body.price;

    console.log("\nFruit Data Received");
    console.log("Fruit Name :", fruitName);
    console.log("Price      :", price);

    res.json({
        message: "Fruit data received successfully"
    });
});

app.listen(PORT, () => {
    console.log(`Server running at http://localhost:${PORT}`);
});
```

# Step 4: Create Client Side HTML

# public/index.html

```html id="vxkmns"
<!DOCTYPE html>
<html>
<head>
    <title>Fruit AJAX App</title>
</head>
<body>

    <h2>Fruit Entry Form</h2>

    <label>Fruit Name:</label>
    <input type="text" id="fruit">

    <br><br>

    <label>Price:</label>
    <input type="number" id="price">

    <br><br>

    <button onclick="sendData()">
        Send Data
    </button>

    <h3 id="result"></h3>

    <script>

        async function sendData() {

            const fruit = document.getElementById('fruit').value;

            const price = document.getElementById('price').value;

            const response = await fetch('/fruit', {

                method: 'POST',

                headers: {
                    'Content-Type': 'application/json'
                },

                body: JSON.stringify({
                    fruit: fruit,
                    price: price
                })
            });

            const data = await response.json();

            document.getElementById('result').innerHTML =
                data.message;
        }

    </script>

</body>
</html>
```
