# backend/server.js

```javascript id="jlwmdd"
const express = require('express');

const cors = require('cors');

const app = express();

const PORT = 3000;

app.use(cors());


// Flight Array

const flights = [

    {
        id: 1,
        flight_name: "Air India",
        source: "Chennai",
        destination: "Delhi",
        price: 6500
    },

    {
        id: 2,
        flight_name: "Indigo",
        source: "Bangalore",
        destination: "Mumbai",
        price: 4500
    },

    {
        id: 3,
        flight_name: "SpiceJet",
        source: "Hyderabad",
        destination: "Kolkata",
        price: 5500
    }
];



// REST API Route

app.get('/flights', (req, res) => {

    res.json(flights);
});



app.listen(PORT, () => {

    console.log(
        `Server running at http://localhost:${PORT}`
    );
});
```

# frontend/src/App.js

```javascript id="jlwmdf"
import React, {
    useEffect,
    useState
} from 'react';

function App() {

    const [flights, setFlights] = useState([]);


    // Fetch Data From API

    useEffect(() => {

        fetch('http://localhost:3000/flights')

            .then((response) =>
                response.json()
            )

            .then((data) => {

                setFlights(data);
            });

    }, []);




    return (

        <div style={{ padding: "20px" }}>

            <h2>
                Flight Details
            </h2>



            <table
                border="1"
                cellPadding="10"

                style={{
                    borderCollapse: "collapse",
                    width: "700px"
                }}
            >

                <thead>

                    <tr>

                        <th>ID</th>
                        <th>Flight Name</th>
                        <th>Source</th>
                        <th>Destination</th>
                        <th>Price</th>

                    </tr>

                </thead>



                <tbody>

                    {
                        flights.map((flight) => (

                            <tr key={flight.id}>

                                <td>
                                    {flight.id}
                                </td>

                                <td>
                                    {flight.flight_name}
                                </td>

                                <td>
                                    {flight.source}
                                </td>

                                <td>
                                    {flight.destination}
                                </td>

                                <td>
                                    {flight.price}
                                </td>

                            </tr>
                        ))
                    }

                </tbody>

            </table>

        </div>
    );
}

export default App;
```
