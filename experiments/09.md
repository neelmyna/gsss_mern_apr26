# Step 5: App.jsx

```javascript id="jlwmen"
import { useState } from 'react';

function App() {

    // Fruit Array

    const fruits = [

        "Apple",
        "Banana",
        "Orange",
        "Grapes",
        "Pineapple",
        "Mango",
        "Papaya"
    ];



    // Search State

    const [search, setSearch] = useState("");



    // Filter Fruits

    const filteredFruits = fruits.filter(
        (fruit) => {

            return fruit
                .toLowerCase()
                .includes(
                    search.toLowerCase()
                );
        }
    );



    return (

        <div style={{ padding: "20px" }}>

            <h2>
                Fruit Search Filter
            </h2>



            <input
                type="text"

                placeholder="Search Fruit"

                value={search}

                onChange={(e) =>
                    setSearch(e.target.value)
                }

                style={{
                    padding: "8px",
                    width: "250px",
                    marginBottom: "20px"
                }}
            />



            <ul>

                {
                    filteredFruits.map(
                        (fruit, index) => (

                        <li key={index}>

                            {fruit}

                        </li>
                    ))
                }

            </ul>

        </div>
    );
}

export default App;
```

# React Features Used

| Feature         | Purpose                 |
| --------------- | ----------------------- |
| `useState`      | Store search text       |
| `onChange`      | Capture user typing     |
| `filter()`      | Filter fruit list       |
| `map()`         | Display filtered list   |
| `includes()`    | Matching logic          |
| `toLowerCase()` | Case insensitive search |

---

```text id="jlwmeu"

      ↓
```
